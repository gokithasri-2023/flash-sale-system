⚙️ FlashSaleService.java — Core Business Logic
// synchronized prevents overselling during Big Billion Day concurrent load @Service public class FlashSaleService { public synchronized OrderResponse placeOrder(OrderRequest req) { Product product = productRepo.findById(req.getProductId()) .orElseThrow(() -> new RuntimeException("Product not found")); // Validate 1: is sale window active? if (!product.getFlashSale().isActive()) return new OrderResponse("FAILED", "Sale not active or expired"); // Validate 2: sufficient stock? if (product.getStock() < req.getQuantity()) return new OrderResponse("FAILED", "Insufficient stock"); // Deduct stock & charge sale price (not original ₹50,000) product.setStock(product.getStock() - req.getQuantity()); double total = product.getFlashSale().getSalePrice() * req.getQuantity(); orderRepo.save(new Order(req.getUserId(), product, total)); productRepo.save(product); return new OrderResponse("SUCCESS", "Total: ₹" + total); } }
